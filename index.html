// Khởi tạo Widget của ERa const eraWidget = new EraWidget(); // Tạo biến lưu trữ hành động (Action) let turnOnLight = null; let turnOffLight = null; // Kết nối với cấu hình trên ERa (cái bảng Actions mà bạn vừa cài đặt) eraWidget.init({ onConfiguration: (configuration) => { // Cấu hình mảng actions: [0] là Hành động 1 (Bật), [1] là Hành động 2 (Tắt) turnOnLight = configuration.actions[0]; turnOffLight = configuration.actions[1]; }, onValues: (values) => { // (Tùy chọn) Chỗ này để nhận trạng thái thực tế từ mạch gửi lên web nếu cần } }); // Hàm xử lý khi bạn bấm công tắc trên giao diện function toggleDevice(deviceId) { // Tìm thẻ hiển thị chữ "Đang bật" / "Đang tắt" const statusText = document.getElementById('status-' + deviceId); // Tìm cái công tắc (checkbox) mà bạn vừa bấm const checkbox = document.querySelector('#device-' + deviceId + ' input[type="checkbox"]'); // Chỉ xử lý cho Đèn Phòng Khách (id = 1) if (deviceId === 1) { if (checkbox.checked) { statusText.innerText = "Đang bật"; // Bắn tín hiệu "Bật" lên máy chủ ERa if (turnOnLight) { turnOnLight.action(); } } else { statusText.innerText = "Đang tắt"; // Bắn tín hiệu "Tắt" lên máy chủ ERa if (turnOffLight) { turnOffLight.action(); } } } // (Sau này bạn có thể viết thêm logic cho device 2, 3, 4 ở đây) }
